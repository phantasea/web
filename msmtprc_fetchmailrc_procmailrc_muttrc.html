.fetchmailrc
# begin of ~/.fetchmailrc
  
defaults
mda "/usr/bin/procmail -d system-user-name"
  
set daemon 600       #每10分钟收取一次邮件
  
# server on GMIAL
poll pop.gmail.com   #邮件服务器IP地址或域名
proto POP3           #POP3协议
port 995
uidl                 #每次只下载新邮件
user "account1"      #在服务器上的用户名
password "xxxxxxxx"  #相应的密码
#keep                #将邮件保存在服务器上
ssl                  #ssl加密
  
# end of ~/.fetchmailrc

.procmailrc
# begin of ~/.procmailrc
  
MAILDIR=$HOME/e-mail
LOGFILE=$HOME/e-mail/.procmaillog
  
# 分类后剩下的信件全扔到这里
DEFAULT=$MAILDIR/in-mail  
  
# Debian Mail List
:0:
* ^(To|Cc|Bcc).*debian-embedded@lists.debian.org  
debian-embedded
:0:
* ^(To|Cc|Bcc).*debian-arm@lists.debian.org  
debian-arm
:0:
* ^(From|To).*debian*  
debian
  
# ucLinux Mail List
:0:
* ^(To|Cc|Bcc).*uclinux-dev@uclinux.org  
uclinux-dev
  
# octave Mail List help-octave@octave.org
:0:
* ^(To|Cc|Bcc).*help-octave@octave.org  
octave
  
# 环球资源
:0:
* ^From.*@globalsources.com
globalsources
  
# 黑名单（垃圾邮件）
:0:
* ^From.mrycn112@163.com
/dev/null
:0:
* ^From.mail@fivebug.com
/dev/null
:0:
* ^From.anny777777@126.com
/dev/null
  
# end of ~/.procmailrc

.muttrc
# begin of ~/.muttrc
  
set sendmail = "/usr/bin/msmtp"
set envelope_from = yes
  
# 邮件文件夹
set folder = ~/e-mail
  
# procmail将从服务器收下来的邮件先进行分类，
# 未被分类的邮件放在这里，
# 告诉mutt, 使用缺省使用这个邮箱
set spoolfile = ~/e-mail/in-mail
  
# 这个是保存草稿文件的
set postponed = "~/e-mail/postponed"
  
# 你发出的邮件保存到那个信箱文件？
# 比如可以像我这样每个月发出的信件放在不同的文件里。
set record="~/e-mail/out-mail-`date +%Y-%m`"    
  
# 这个是保存已经阅读的邮件的,
# 现在我用procmail自动分类邮件了，详情见~/.procmialrc文件
# set mbox = "~/e-mail/in-mail-`date +%Y-%m`"
  
# 让mutt监视下面几个邮箱，并随时报告新邮件
mailboxes "=in-mail"
mailboxes "=debian-embedded"
mailboxes "=debian-arm"
mailboxes "=debian"
mailboxes "=uclinux-dev"
mailboxes "=globalsources"
set check_new = yes
set timeout = 600
  
# 设置自己的邮件头
# 缺省邮箱
send-hook . 'my_hdr from: account1<account1@gmail.com>  
send-hook . 'my_hdr Reply-To: account1<account1@gmail.com>
# 给公司同事的邮件用公司的邮箱
send-hook '~t .*@tsinghua-sz\.edu\.cn$' 'my_hdr from: account2<account2@tsinghua-sz.edu.cn>  
send-hook '~t .*@tsinghua-sz\.edu\.cn$' 'my_hdr Reply-To: account2<account2@tsinghua-sz.edu.cn>
# 订阅邮件列表的专用邮箱
send-hook '~t uclinux-dev@uclinux\.org$' 'my_hdr from: account3<account3@gmail.com>  
send-hook '~t uclinux-dev@uclinux\.org$' 'my_hdr Reply-To: account3<account3@gmail.com>
  
# 你的终端支持哪一种编码的显示？这个必须和你的终端编码一样。
set charset= "gbk"                      
  
# mutt显示日期为中文
set locale = "zh_CN.GBK"                
  
# 看信时把所有不是utf-8的都做gbk处理
charset-hook !utf-8 gbk
  
# 在回信时引用原文是否加入原文的邮件头
set header = no                          
  
# 当你要把信件存到另一个信箱，而这个信箱已经存在时，是否提示附加？
set noconfirmappend
  
# 告诉Mutt已经订阅了那些邮件列表(mailing-list). 按L可以回复邮件列表
subscribe uclinux-dev@uclinux.org
subscribe debian-arm@lists.debian.org
subscribe debian-embedded@lists.debian.org
subscribe help-octave@octave.org
  
# 可以设置为 yes, no, ask-yes, 或者 ask-no. 这是说，当你按q退出时，是否提示你(ask-yes,ask-no)，
# 还是直接就退出了(yes)，还是根本不理你(no)。
set quit = ask-yes
  
# 退出时，是否提示你把信件从spool移动到mbox文件。
set move = no
  
# 当Mutt用thread方式显示时，是否用纯ascii表示树状列表。
set ascii_chars = yes
  
# 回信时是否包含原文。
set include = yes
  
# 回信的引文之前插入那个符号？
set indent_str = "> "
  
# 用gvim作编辑器
set editor = vim
  
# 打分
# 新信件+4分,主题包含“通知”的+2，主题包含 “Circulation” +3, 已经标记
# 删除的 -5，上次没有读的 +1，包含 “believe”的 -10(垃圾广告！)。
score "~N" +4
score "~s 通知" +2
score "~s Circulation" +3  
score "~D" -5
score "~O" +1
score "~s believe" -10
# 排序方式。
set sort = reverse-date
set sort_aux = score
  
# 按"r"回信时，是(no)/否(yes)提示你输入回信地址和主题。
set fast_reply = no
  
# 如果翻到最后，再按往下翻也不会翻到下一封信件
set pager_stop = yes
  
# 当你按"t"或者"D"之类的标记操作时，是否自动把光标移动到下一封信件。
set resolve = yes
  
# " "(空): 表示这封邮件不是给你的，也就是说 To: 和 Cc: Bcc: 都没有你的地址，很多转发的邮件，
#          未确认的邮件列表邮件，垃圾邮件都有这个特征。
# +: 表示你是收件人(To: 是你的地址之一)，而且是唯一的收件人。
# T: 表示你是收件人(To: 包括了你的地址之一)，但是你不是唯一的收件人。这是一封群体信件。
# C: 表示你的地址出现在 CC:，但是你不是唯一的被抄送的人。
# F: 表示这封邮件是你发出去的。
# L: 表示这是一封你已经加入的邮件列表寄来的。
set to_chars = " +TCFL"
  
# 地址簿
source ~/e-mail/mutt.alias
  
# 设置了这个以后，在mutt中按"a"会自动把联系人加到这个文件里面
set alias_file = ~/e-mail/mutt.alias
  
set send_charset="us-ascii:iso-8859-1:gb2312:gbk:utf-8"
  
# 外部程序退出时，是否要求用户按一个键才返回。这在察看某些shell命令输出时是比要的，
# 否则它们一下就消失了。
set wait_key = yes
  
# 是否把邮件头也放在编辑器里可以修改？
set edit_headers = no
  
# 当你在用 pager 看信件时，在 index 留出多少行显示邮件列表？
set pager_index_lines = 8        
  
# 这是一个格式字符串，用来控制你的index的列表显示。它的缺省定义是  
# set index_format = "%4C %Z %{%b %d} %-15.15L (%4l) %s"  
set index_format = "%Z %2C |%{%b%d日} %-15.15L (%5lk) %s"
  
# 我的out-mail.*(发件箱)里只显示 To: Subject: Date: 三项邮件头
# 而在其它信箱就显示 From: Subject: X-Mailer: Date: 这几项。
folder-hook . "ignore *; unignore from to subject X-mailer date"
folder-hook out-mail.* "ignore *; unignore to subject date"  
  
# 指定邮件头显示的顺序
hdr_order X-Mailer User-Agent date from to subject  
  
# 在mutt中调用lynx浏览html附件
auto_view text/html
  
# 以下是界面的配色
  
# end of ~/.muttrc
