text/html; w3m -I %{charset} -T text/html ; nametemplate=%s.html; copiousoutput;
application/octet-stream;   xarchiver %s; test=test -n "$DISPLAY"
video/*;         mplayer %s > /dev/null
audio/*;         mplayer %s > /dev/null
image/*;         feh -. %s > /dev/null
application/pdf; FoxitReader %s > /dev/null

word文档和excle 我转成html 来查看，给你一份脚本，随手写的：
代码:
#!/usr/bin/env bash
# 需要 wvHtml w3m
source=$1
target=`mktemp`

if echo $source | grep -q "doc"; then
    wvHtml $source $target.html
    w3m $target.html
    rm -f $target*
elif echo $source | grep -q "xls"; then
    xlhtml $source > $target.html
    w3m $target.html
    rm -f $target*
fi
