    你附件中关于切换视图的那几个功能很好用。我稍稍做了一下改进，就是用<S-up>
    <S-down>来toggle上面的工具栏。
    用<C-up>/<C-down>来toggle下面的工具栏。取义是shift键在contrl键的上部，与屏幕相对应。

    以下是我修改的部分，供参考:
    nnoremap <silent> <S-Up> :js ToggleUp('up')<CR>
    nnoremap <silent> <S-Down> :js ToggleUp('down')<CR>
    nnoremap <silent> <C-Up> :js ToggleBottom('up')<CR>
    nnoremap <silent> <C-Down> :js ToggleBottom('down')<CR>


    " Toggle the upside bars
    set guioptions=brN
    set showtabline=2
    js <<EOF
    var upBar=1;
    ToggleUp = function(dir) {
        if (upBar == 1) {
            if (dir == 'up') {
                liberator.execute('set showtabline=0');
                upBar = 2
            } else if ( dir == 'down') {
                liberator.execute('set guioptions+=mT');
                upBar = 0
            }
        } else if (upBar == 2) {
            if (dir == 'down') {
                liberator.execute('set showtabline=2');
                upBar = 1
            }
        } else if (upBar == 0) {
            if (dir == 'up') {
                upBar = 1
                liberator.execute('set guioptions-=mT');
            }
        }
    }
    EOF
    " Toggle the bottom view area
    set laststatus=0
    js <<EOF
    var BottomBar=1;
    ToggleBottom = function(dir) {
        if (BottomBar == 1) {
            if (dir == 'up') {
                liberator.execute('set laststatus=2');
                BottomBar = 2
            } else if ( dir == 'down') {
                //hidden the windows task bar
                window.maximize()
                BottomBar = 0
            }
        } else if (BottomBar == 0) {
            if (dir == 'up') {
                window.restore()
                BottomBar = 1
            }
        } else if (BottomBar == 2) {
            if (dir == 'down') {
                liberator.execute('set laststatus=0');
                BottomBar = 1
            }
        }
    }
    EOF
